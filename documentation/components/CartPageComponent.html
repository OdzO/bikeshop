import { Component, ViewChild } from '@angular/core';
import { MatTable } from '@angular/material/table';
import { Product } from 'src/app/interfaces/product';
import { CartService } from 'src/app/services/cart.service';

@Component({
  selector: 'app-cart-page',
  templateUrl: './cart-page.component.html',
  styleUrls: ['./cart-page.component.scss']
})
export class CartPageComponent {

  constructor(public cart: CartService) { }

  @ViewChild(MatTable) table!: MatTable<Product>;
  displayedColumns: string[] = ['name', 'price', 'action'];
}

<div id="cartContainer">
  <div id="cartHeader">Your cart:</div>
  <table mat-table [dataSource]="cart.getCart()" class="mat-elevation-z8" aria-label="Table of products">
    <ng-container matColumnDef="name">
      <th mat-header-cell *matHeaderCellDef id="col-1">Name</th>
      <td mat-cell *matCellDef="let prod">{{ prod.name }}</td>
      <td mat-footer-cell *matFooterCellDef>Total</td>
    </ng-container>

    <ng-container matColumnDef="price">
      <th mat-header-cell *matHeaderCellDef id="col-1">Price</th>
      <td mat-cell *matCellDef="let prod">
        <span *ngIf="!prod.sale">{{ prod.price | currency }}</span>
        <span *ngIf="prod.sale">{{ prod.price | sale: prod.sale | currency }}</span>
      </td>
      <td mat-footer-cell *matFooterCellDef>{{ cart.getCartTotal() | currency }}</td>
    </ng-container>

    <ng-container matColumnDef="action">
      <th mat-header-cell *matHeaderCellDef id="col-4"></th>
      <td mat-cell *matCellDef="let prod">
        <button mat-icon-button color="basic" aria-label="Example icon button with a cancel icon" (click)="cart.removeFromCart(prod)">
          <mat-icon>cancel</mat-icon>
        </button>
      </td>
      <td mat-footer-cell *matFooterCellDef></td>
    </ng-container>

    <tr mat-header-row *matHeaderRowDef="displayedColumns"></tr>
    <tr mat-row *matRowDef="let row; columns: displayedColumns"></tr>
    <tr mat-footer-row *matFooterRowDef="displayedColumns"></tr>
  </table>
</div>
